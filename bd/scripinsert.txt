
CREATE DATABASE incidencias;
USE incidencias;

CREATE TABLE Areas(
	id_areas               INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_areas      VARCHAR(100) NOT NULL,
	estado_areas           CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Areas_id_areas_INX ON Areas(id_areas);
CREATE UNIQUE INDEX Areas_descripcion_areas_INX ON Areas(descripcion_areas);

CREATE TABLE CategoriaBien(
	id_catb               INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_catb      VARCHAR(100) NOT NULL,
	estado_catb           CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX CategoriaBien_id_catb_INX ON CategoriaBien(id_catb);
CREATE UNIQUE INDEX CategoriaBien_descripcion_catb_INX ON CategoriaBien(descripcion_catb);

CREATE TABLE CategoriaIncidencia(
	id_catinc             INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_catinc    VARCHAR(100) NOT NULL,
	estado_catinc         CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX CategoriaIncidencia_id_catinc_INX ON CategoriaIncidencia(id_catinc);
CREATE UNIQUE INDEX CategoriaIncidencia_descripcion_catinc_INX ON CategoriaIncidencia(descripcion_catinc);

CREATE TABLE Distrito(
	id_distrito           INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_distrito  VARCHAR(100) NOT NULL,
	estado_distrito       CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Distrito_id_distrito_INX ON Distrito(id_distrito);
CREATE UNIQUE INDEX Distrito_descripcion_distrito_INX ON Distrito(descripcion_distrito);

CREATE TABLE Estado(
	id_estado             INTEGER PRIMARY KEY  NOT NULL auto_increment,
	descripcion_estado    VARCHAR(100) NOT NULL,
	estado_estado         CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Estado_id_estado_INX ON Estado(id_estado);
CREATE UNIQUE INDEX Estado_descripcion_estado_INX ON Estado(descripcion_estado);

CREATE TABLE Marca(
	id_marca              INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_marca     VARCHAR(100) NOT NULL,
	estado_marca          CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Marca_Marca_INX ON Marca(id_marca);
CREATE UNIQUE INDEX Marca_descripcion_marca_INX ON Marca(descripcion_marca);

CREATE TABLE Modelo(
	id_modelo             INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_modelo    VARCHAR(100) NOT NULL,
	estado_modelo         CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Modelo_id_modelo_INX ON Modelo(id_modelo);
CREATE UNIQUE INDEX Modelo_descripcion_modelo_INX ON Modelo(descripcion_modelo);

CREATE TABLE Prioridad(
	id_prioridad           INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_prioridad  VARCHAR(100) NOT NULL,
	estado_prioridad       CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Prioridad_id_prioridad_INX ON Prioridad(id_prioridad);
CREATE UNIQUE INDEX Prioridad_descripcion_prioridad_INX ON Prioridad(descripcion_prioridad);

CREATE TABLE TipoUsuario(
	id_tipousuario           INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_tipousuario  VARCHAR(100) NOT NULL,
	estado_tipousuario       CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX TipoUsuario_id_tipousuario_INX ON TipoUsuario(id_tipousuario);
CREATE UNIQUE INDEX TipoUsuario_descripcion_tipousuario_INX ON TipoUsuario(descripcion_tipousuario);

CREATE TABLE AsuntoIncidencia(
	id_asuinc             INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_asuinc    VARCHAR(100) NOT NULL,
	estado_asuinc         CHAR(1) NOT NULL,
	id_catinc             INTEGER NOT NULL,
	FOREIGN KEY (id_catinc) REFERENCES CategoriaIncidencia(id_catinc)
);

CREATE UNIQUE INDEX AsuntoIncidencia_id_asuinc_INX ON AsuntoIncidencia(id_asuinc);
CREATE UNIQUE INDEX AsuntoIncidencia_descripcion_asuinc_INX ON AsuntoIncidencia(descripcion_asuinc);

CREATE TABLE Bien(
	id_bien               INTEGER PRIMARY KEY NOT NULL auto_increment,
	codigo_bien	          VARCHAR(100) NOT NULL,
	serie_bien	          VARCHAR(100) NOT NULL,
	descripcion_bien      VARCHAR(100) NOT NULL,
	precio_bien           DECIMAL(18,2) NOT NULL,
	fecha_bien            DATE NOT NULL,
	hora_bien             TIME NOT NULL,
	cantidad_bien         INTEGER(11) NOT NULL,
	estado_bien           CHAR(1) NOT NULL,
	id_modelo             INTEGER NOT NULL,
	id_marca              INTEGER NOT NULL,
	id_catb               INTEGER NOT NULL,
	FOREIGN KEY (id_modelo) REFERENCES Modelo(id_modelo),
	FOREIGN KEY (id_marca) REFERENCES Marca(id_marca),
	FOREIGN KEY (id_catb) REFERENCES CategoriaBien(id_catb)
);

CREATE UNIQUE INDEX Bien_id_bien_INX ON Bien(id_bien);
CREATE UNIQUE INDEX Bien_codigo_bien_INX ON Bien(codigo_bien);
CREATE UNIQUE INDEX Bien_serie_bien_INX ON Bien(serie_bien);
CREATE UNIQUE INDEX Bien_descripcion_bien_INX ON Bien(descripcion_bien);

CREATE TABLE Persona(
	id_persona            INTEGER PRIMARY KEY NOT NULL auto_increment,
	nombres_persona       VARCHAR(100) NOT NULL,
	apepat_persona        VARCHAR(20) NOT NULL,
	apeat_persona         VARCHAR(20) NOT NULL,
	dni_persona           CHAR(8) NOT NULL,
	telf_persona          CHAR(10) NOT NULL,
	telm_persona          CHAR(10) NOT NULL,
	correo_persona        CHAR(20) NOT NULL,
	direccion_persona     VARCHAR(100) NOT NULL,
    sueldo_persona        DECIMAL(18, 2) NOT NULL,
	estado_persona        CHAR(1) NOT NULL,
	id_distrito           INTEGER NOT NULL,
	FOREIGN KEY (id_distrito) REFERENCES Distrito(id_distrito)
);

CREATE UNIQUE INDEX Persona_id_persona_INX ON Persona(id_persona);
CREATE UNIQUE INDEX Persona_dni_persona_INX ON Persona(dni_persona);

CREATE TABLE Nivel(
	id_nivel                INTEGER PRIMARY KEY NOT NULL auto_increment,
    nivel_nivel             CHAR(2) NOT NULL,
	descripcion_nivel       VARCHAR(100) NOT NULL,
	estado_nivel            CHAR(1) NOT NULL
);

CREATE UNIQUE INDEX Nivel_id_nivel_INX ON Nivel(id_nivel);
CREATE UNIQUE INDEX Nivel_descripcion_nivel_INX ON Nivel(descripcion_nivel);

CREATE TABLE Usuario(
	id_usuario            INTEGER PRIMARY KEY NOT NULL auto_increment,
	usu_usuario           VARCHAR(20) NOT NULL,
	pas_usuario           VARCHAR(20) NOT NULL,
	estado_usuario        CHAR(1) NOT NULL,	
	id_areas              INTEGER NOT NULL,
	id_persona            INTEGER NOT NULL,
	id_tipousuario        INTEGER NOT NULL,
    id_nivel              INTEGER NOT NULL,
	FOREIGN KEY (id_areas) REFERENCES Areas(id_areas),
	FOREIGN KEY (id_persona) REFERENCES Persona(id_persona),
	FOREIGN KEY (id_tipousuario) REFERENCES TipoUsuario(id_tipousuario),
    FOREIGN KEY (id_nivel) REFERENCES Nivel(id_nivel)
);

CREATE UNIQUE INDEX Usuario_id_usuario_INX ON Usuario(id_usuario);
CREATE UNIQUE INDEX Usuario_usu_usuario_INX ON Usuario(usu_usuario);

CREATE TABLE Incidencia(
	id_incidencia          INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_incidencia VARCHAR(100) NOT NULL,
	fini_incidencia        DATE NOT NULL,
	hini_incidencia        TIME NOT NULL,
	ffin_incidencia        DATE,
	hfin_incidencia        TIME,
	canal_incidencia       CHAR(20) NOT NULL,
    cliente_incidencia     VARCHAR(100) NOT NULL,
    responsable_incidencia VARCHAR(100),
    tresponsable_incidencia VARCHAR(100),
    tipo_incidencia VARCHAR(20) NOT NULL,
	estado_incidencia      CHAR(1) NOT NULL,
	id_prioridad           INTEGER NOT NULL,
	id_estado              INTEGER NOT NULL,
	id_asuinc              INTEGER NOT NULL,
	id_bien                INTEGER NOT NULL,
	id_usuario             INTEGER NOT NULL,
	FOREIGN KEY (id_prioridad) REFERENCES Prioridad(id_prioridad),
	FOREIGN KEY (id_estado) REFERENCES Estado(id_estado),
	FOREIGN KEY (id_asuinc) REFERENCES AsuntoIncidencia(id_asuinc),
	FOREIGN KEY (id_bien) REFERENCES Bien(id_bien),
	FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);

CREATE TABLE Diagnostico(
	id_diagnostico           INTEGER PRIMARY KEY NOT NULL auto_increment,
	descripcion_diagnostico  VARCHAR(100) NOT NULL,
	fecha_diagnostico        DATE NOT NULL,
	hora_diagnostico         TIME NOT NULL,
	id_incidencia            INTEGER NOT NULL,
	FOREIGN KEY (id_incidencia) REFERENCES Incidencia(id_incidencia)
);
